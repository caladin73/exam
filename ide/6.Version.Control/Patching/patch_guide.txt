Patches step by step:
// go into folder a and:
git init
git add .
git commit -m "first"
git log 
//copy hash
git format-patch -1 e2391119006a67cc64e1bfd2a37d38e8d4005076 -o patches

//copy name and apply into new folder b
git init 
git am 0001-first.patch

//back again, edit file
git add .
git commit -m "new line"
git format-patch -1 27c28bc32049fc7c228d0505e7589153d7fa9ede -o patches
//copy file to a
git am 0001-change.patch
